72 static void discharge (uint8_t which_AIN)
73 {
74  if (EC_VIN != which_AIN && EC_VOUT != which_AIN) {
75  APP_ERROR_HANDLER(LADYBUG_ERROR_INVALID_COMMAND);
76  }
77  uint32_t pin_number;
78  //set the GPIO based on whether the current ADC reading is for the VIN or VOUT
79  if (EC_VIN == which_AIN) {
80  pin_number = m_EC_VIN_FET;

82  else {
83  pin_number = m_EC_VOUT_FET;

85  //configure the GPIO for output
86  nrf_gpio_cfg_output(pin_number);

89  //open the FET's gate so the ADC picks up an accurate measurement
90  nrf_gpio_pin_clear(pin_number);

112 static void get_EC_reading(int16_t *p_EC) {
113  if (p_EC == NULL){ //Shouldn't be passing in a null pointer given the EC Vin and Vout values are planned to be stored at this memory location.
114  APP_ERROR_HANDLER(LADYBUG_ERROR_NULL_POINTER);

116  SEGGER_RTT_WriteString(0,"---> IN get_EC_reading\n");

118  SEGGER_RTT_printf(0,"EC_VGND: %d 0X%x\n",VGND,VGND);
119  //EC VIN and EC VOUT have a rectifier step in which there is a FET that stabilizes the rectification by discharging the cap to prevent an upward drift..

138 void ladybug_update_calibration_value(control_enum_t command, int solutionValue){
139  SEGGER_RTT_printf(0,"---> in ladybug_update_calibration_value. command: %d, solution value: %d\n",command,solutionValue);
140  if (command != calibratePH4 && command != calibratepH7 && command != calibrateEC1 && command != calibrateEC2){
141  APP_ERROR_HANDLER(LADYBUG_ERROR_INVALID_COMMAND);
142  }
143  //pH calibration comes from a simple reading of the pH AIN
144  //there is no additional calibration values that need to be stored since calibration is fixed on using the two points: pH4 and pH7.
145  if (command == calibratePH4 || command == calibratepH7) {
146  int16_t pH_value = get_pH_reading();
147  if (command == calibratePH4){
148  m_storeCalibrationValues.calValues.pH4_mV = pH_value;
149  SEGGER_RTT_WriteString(0,"Calibrated pH4\n");
150  }else {
151  m_storeCalibrationValues.calValues.pH7_mV = pH_value;
152  SEGGER_RTT_WriteString(0,"Calibrated pH7\n");
153  }
154  }else {
155  //EC calibration can use one or two points. The calibration's solution values are stored in EC1solution and EC2solution. The readings from the probe
156  //taken when the probe is submerged in either the EC1solution or EC2solution is EC1[] or EC2[].
157  int16_t EC_VIN_and_VOUT_mV[2];
158  // Read the AIN values assigned for the EC Vin and EC Vout values. Which is read depends on which calibration solution the
159  // probe is in. The user of the client has chosen either EC1 or EC2. What comes over is the EC 1 or 2 calibration solution
160  // value. This will be stored as well as the probe values that are read from which the EC can be calculated on the client.
161  get_EC_reading(EC_VIN_and_VOUT_mV); //the first element is VIN, the second is VOUT. EC calculation happens on the client
162  if (command == calibrateEC1){
163  SEGGER_RTT_WriteString(0,"...setting EC1 values...\n");
164  m_storeCalibrationValues.calValues.EC1solution = solutionValue;
165  for (int i=0;i<2;i++) {
166  m_storeCalibrationValues.calValues.EC1_mV[i] = EC_VIN_and_VOUT_mV[i];
167  }
168  }else { //calibrate EC2
169  SEGGER_RTT_WriteString(0,"...setting EC2 values...\n");
170  m_storeCalibrationValues.calValues.EC2solution = solutionValue;
171  for (int i=0;i<2;i++) {
172  m_storeCalibrationValues.calValues.EC2_mV[i] = EC_VIN_and_VOUT_mV[i];
173  }
174  }
175  print_out_calibration_values();
176  }
177  //write the reading (and the rest that in the hydro data) to flash.
178  m_write_calibration_values = true;
179 }

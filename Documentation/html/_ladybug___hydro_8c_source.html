53 static void discharge (uint8_t which_AIN)
54 {
55  uint32_t pin_number;
56  //set the GPIO based on whether the current ADC reading is for the VIN or VOUT
57  if (EC_VIN == which_AIN) {
58  pin_number = EC_VIN_FET;
59  }
60  else {
61  pin_number = EC_VOUT_FET;
62  }
63  //configure the GPIO for output
64  nrf_gpio_cfg_output(pin_number);
65  //discharge the cap by setting the GPIO to HIG
66  nrf_gpio_pin_set(pin_number);
67  //open the FET's gate so the ADC picks up an accurate measurement
68  nrf_gpio_pin_clear(pin_number);

128 void ladybug_update_calibration_value(control_enum_t command, int solutionValue){
129  SEGGER_RTT_printf(0,"---> in ladybug_update_calibration_value. command: %d, solution value: %d\n",command,solutionValue);
130  //pH calibration comes from a simple reading of the pH AIN
131  //there is no additional calibration values that need to be stored since calibration is fixed on using the two points: pH4 and pH7.
132  if (command == calibratePH4 || command == calibratepH7) {
133  int16_t pH_value = get_pH_reading();
134  if (command == calibratePH4){
135  m_storeCalibrationValues.calValues.pH4_mV = pH_value;
136  SEGGER_RTT_WriteString(0,"Calibrated pH4\n");
137  }else {
138  m_storeCalibrationValues.calValues.pH7_mV = pH_value;
139  SEGGER_RTT_WriteString(0,"Calibrated pH7\n");
140  }
141  }else {
142  //EC calibration can use one or two points. The calibration's solution values are stored in EC1solution and EC2solution. The readings from the probe
143  //taken when the probe is submerged in either the EC1solution or EC2solution is EC1[] or EC2[].
144  int16_t EC_VIN_and_VOUT_mV[2];
145  // Read the AIN values assigned for the EC Vin and EC Vout values. Which is read depends on which calibration solution the
146  // probe is in. The user of the client has chosen either EC1 or EC2. What comes over is the EC 1 or 2 calibration solution
147  // value. This will be stored as well as the probe values that are read from which the EC can be calculated on the client.
148  get_EC_reading(EC_VIN_and_VOUT_mV); //the first element is VIN, the second is VOUT. EC calculation happens on the client
149  if (command == calibrateEC1){
150  SEGGER_RTT_WriteString(0,"...setting EC1 values...\n");
151  m_storeCalibrationValues.calValues.EC1solution = solutionValue;
152  for (int i=0;i<2;i++) {
153  m_storeCalibrationValues.calValues.EC1_mV[i] = EC_VIN_and_VOUT_mV[i];
154  }
155  }else { //calibrate EC2
156  SEGGER_RTT_WriteString(0,"...setting EC2 values...\n");
157  m_storeCalibrationValues.calValues.EC2solution = solutionValue;
158  for (int i=0;i<2;i++) {
159  m_storeCalibrationValues.calValues.EC2_mV[i] = EC_VIN_and_VOUT_mV[i];
160  }
161  }
162  print_out_calibration_values();
163  }
164  //write the reading (and the rest that in the hydro data) to flash.
165  m_write_calibration_values = true;
166 }

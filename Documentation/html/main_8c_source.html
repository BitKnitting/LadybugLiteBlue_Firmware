304 static void on_ble_evt(ble_evt_t * p_ble_evt)
305 {
306  uint32_t err_code;
307  static ble_gap_evt_auth_status_t m_auth_status;
308  static ble_gap_master_id_t p_master_id;
309  static ble_gap_sec_keyset_t keys_exchanged;
310  SEGGER_RTT_WriteString(0,"\n--> IN on_ble_evt (a BLE event has come into main.c\n");
311 
312  switch (p_ble_evt->header.evt_id)
313  {
314  case BLE_GAP_EVT_CONNECTED:
315  SEGGER_RTT_WriteString(0,"\n... BLE_GAP_EVT_CONNECTED\n");
316  m_conn_handle = p_ble_evt->evt.gap_evt.conn_handle;
317 
318  break;
319 
320  case BLE_GAP_EVT_DISCONNECTED:
321  SEGGER_RTT_WriteString(0,"\n... BLE_GAP_EVT_DISCONNECTED\n");
322  m_conn_handle = BLE_CONN_HANDLE_INVALID;
323  advertising_start();
324  break;
325 
326  case BLE_GAP_EVT_SEC_PARAMS_REQUEST:
327  SEGGER_RTT_WriteString(0,"\n... BLE_GAP_EVT_SEC_PARAMS_REQUEST\n");
328  err_code = sd_ble_gap_sec_params_reply(m_conn_handle,
329  BLE_GAP_SEC_STATUS_SUCCESS,
330  &m_sec_params,&keys_exchanged);
331  APP_ERROR_CHECK(err_code);
332  break;
333 
334  case BLE_GATTS_EVT_SYS_ATTR_MISSING:
335  SEGGER_RTT_WriteString(0,"\n... BLE_GATTS_EVT_SYS_ATTR_MISSING\n");
336  err_code = sd_ble_gatts_sys_attr_set(m_conn_handle, NULL, 0,BLE_GATTS_SYS_ATTR_FLAG_USR_SRVCS);
337  APP_ERROR_CHECK(err_code);
338  break;
339 
340  case BLE_GAP_EVT_AUTH_STATUS:
341  SEGGER_RTT_WriteString(0,"\n... BLE_GAP_EVT_AUTH_STATUS\n");
342  m_auth_status = p_ble_evt->evt.gap_evt.params.auth_status;
343  break;
344 
345  case BLE_GAP_EVT_SEC_INFO_REQUEST:
346  //p_enc_info = keys_exchanged.keys_central.p_enc_key
347  SEGGER_RTT_WriteString(0,"\n... BLE_GAP_EVT_SEC_INFO_REQUEST\n");
348  if (p_master_id.ediv == p_ble_evt->evt.gap_evt.params.sec_info_request.master_id.ediv)
349  {
350  err_code = sd_ble_gap_sec_info_reply(m_conn_handle, &keys_exchanged.keys_central.p_enc_key->enc_info, &keys_exchanged.keys_central.p_id_key->id_info, NULL);
351  APP_ERROR_CHECK(err_code);
352  p_master_id.ediv = p_ble_evt->evt.gap_evt.params.sec_info_request.master_id.ediv;
353  }
354  else
355  {
356  // No keys found for this device
357  err_code = sd_ble_gap_sec_info_reply(m_conn_handle, NULL, NULL,NULL);
358  APP_ERROR_CHECK(err_code);
359  }
360  break;
361 
362  case BLE_GAP_EVT_TIMEOUT:
363  SEGGER_RTT_WriteString(0,"\n... BLE_GAP_EVT_TIMEOUT\n");
364  if (p_ble_evt->evt.gap_evt.params.timeout.src == BLE_GAP_TIMEOUT_SRC_ADVERTISING)
365  {
366  // Go to system-off mode (this function will not return; wakeup will cause a reset)
367  err_code = sd_power_system_off();
368  APP_ERROR_CHECK(err_code);
369  }
370  break;
371 
372  default:
373  SEGGER_RTT_WriteString(0,"\n... Not handling a BLE event here.\n");
374  // No implementation needed.
375  break;
376  }
377 }

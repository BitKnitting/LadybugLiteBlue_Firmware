466  * calibration values. This function figures out what block handle has been initialized based on what info has been requested to read, reads the info from flash, then writes
467  * the info to the p_bytes_to_read memory buffer.
468  * @param data_to_read let the function know what type of data to read
469  * @param p_bytes_to_read give the function a buffer to write the data after reading from flash
470  * @sa LADYBUG_ERROR_UNSURE_WHAT_DATA_TO_READ
471  */

473  m_mypstorage_wait_flag = 1;

534 static void flash_init() {

536  pstorage_handle_t handle; //used to access the chunk-o-flash requested when registering
537  //First thing is to initialize pstorage
538  uint32_t err_code = pstorage_init();
539  //if initialization was not successful, the error routine will be called and the code won't proceed.
540  APP_ERROR_CHECK(err_code);
541  //Next is to register amount of flash needed. The smallest amount that can be requested is 16 bytes. I'm requesting
542  //32 bytes because this is the number of bytes needed for plant_info. Two blocks are used...one for plant info and one for hydro values. I must make a

544  pstorage_param.block_size = BLOCK_SIZE;
545  //request three blocks - one will be for pH, one for plant_info, and one for device name
546  pstorage_param.block_count = 3;

549  err_code = pstorage_register(&pstorage_param, &handle);

553  pstorage_block_identifier_get(&handle,1,&m_block_plantInfo_store_handle);
554  pstorage_block_identifier_get(&handle,2,&m_block_device_name_store_handle);
555 
556  APP_ERROR_CHECK(err_code);

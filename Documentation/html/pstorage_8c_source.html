385 static void state_data_erase_swap_entry_run(void)
386 {
387  m_flags &= ~MASK_TAIL_SWAP_DONE;
388 
389  const cmd_queue_element_t * p_cmd = &m_cmd_queue.cmd[m_cmd_queue.rp];
390  const pstorage_block_t cmd_block_id = p_cmd->storage_addr.block_id;
391 
392  const uint32_t clear_start_page_id = cmd_block_id / PSTORAGE_FLASH_PAGE_SIZE;
393  m_current_page_id = clear_start_page_id;
394 
395  // @note: No need to include p_cmd->offset when calculating clear_end_page_id as:
396  // - clear API does not include offset parameter
397  // - update and store APIs are limited to operate on single block boundary thus the boolean
398  // clause ((m_head_word_size == 0) && is_more_than_one_page) below in this function will never
399  // evaluate as true as if is_more_than_one_page == true m_head_word_size is always != 0
400  const uint32_t clear_end_page_id = (cmd_block_id + p_cmd->size - 1u) /
401  PSTORAGE_FLASH_PAGE_SIZE;
402 
403  if (clear_start_page_id == clear_end_page_id)
404  {
405  m_flags |= MASK_SINGLE_PAGE_OPERATION;
406  }
407  else
408  {
409  m_flags &= ~MASK_SINGLE_PAGE_OPERATION;
410  }
411 
412  if ((m_head_word_size == 0) && !(m_flags & MASK_SINGLE_PAGE_OPERATION))
413  {
414  // No head restore required and clear/update area is shared by multiple flash pages, which
415  // means the current flash page does not have any tail area to restore. You can proceed with
416  // data page erase directly as no swap is needed for the current flash page.
417  swap_sub_state_state_change(STATE_ERASE_DATA_PAGE);
418  }
419  else
420  {
421  swap_sub_state_state_change(STATE_ERASE_SWAP);
422  }
423 }

626 static uint32_t cmd_queue_enqueue(uint8_t opcode,
627  pstorage_handle_t * p_storage_addr,
628  uint8_t * p_data_addr,
629  pstorage_size_t size,
630  pstorage_size_t offset)
631 {
632  uint32_t err_code;
633 
634  if (m_cmd_queue.count != PSTORAGE_CMD_QUEUE_SIZE)
635  {
636  // Enqueue the command if it the queue is not full.
637  uint32_t write_index = m_cmd_queue.rp + m_cmd_queue.count;
638 
639  if (write_index >= PSTORAGE_CMD_QUEUE_SIZE)
640  {
641  write_index -= PSTORAGE_CMD_QUEUE_SIZE;
642  }
643 
644  m_cmd_queue.cmd[write_index].op_code = opcode;
645  m_cmd_queue.cmd[write_index].p_data_addr = p_data_addr;
646  m_cmd_queue.cmd[write_index].storage_addr = (*p_storage_addr);
647  m_cmd_queue.cmd[write_index].size = size;
648  m_cmd_queue.cmd[write_index].offset = offset;
649 
650  m_cmd_queue.count++;
651 
652  if (m_state == STATE_IDLE)
653  {
654  cmd_process();
655  }
656 
657  err_code = NRF_SUCCESS;
658  }
659  else
660  {
661  err_code = NRF_ERROR_NO_MEM;
662  }
663 
664  return err_code;
665 }

811 static void head_restore_state_run(void)
812 {
813  if (!(m_flags & MASK_FLASH_API_ERR_BUSY))
814  {
815  if (is_tail_data_page_swap_required())
816  {
817  // Additional data page needs to be swapped for tail section as we are clearing a block,
818  // which is shared between 2 flash pages.
819 
820  // Adjust variables to ensure correct state transition path is taken after the tail
821  // section swap has completed.
822  m_head_word_size = 0;
823  m_flags |= MASK_TAIL_SWAP_DONE;
824 
825  swap_sub_state_state_change(STATE_ERASE_SWAP);
826  }
827  else if (is_page_erase_required())
828  {
829  // Additional page erase operation is required.
830 
831  // Adjust variable to ensure correct state transition path is taken after the additional
832  // page erase operation has completed.
833  m_head_word_size = 0;
834  swap_sub_state_state_change(STATE_ERASE_DATA_PAGE);
835  }
836  else if (m_tail_word_size != 0)
837  {
838  // Proceed with restoring tail from swap to data page.
839  swap_sub_state_state_change(STATE_RESTORE_TAIL);
840  }
841  else
842  {
843  // Swap statemachine execution end reached.
844  swap_sub_sm_exit_action_run();
845  }
846  }
847  else
848  {
849  // As operation request was rejected by the flash API reissue the request.
850  swap_sub_state_err_busy_process();
851  }
852 }

876 static void data_page_erase_state_run(void)
877 {
878  if (!(m_flags & MASK_FLASH_API_ERR_BUSY))
879  {

887  {
888  // Additional page erase operation is required.
889  swap_sub_state_state_change(STATE_ERASE_DATA_PAGE);
890  }
891  else if (m_tail_word_size != 0)
892  {
893  if (!(m_flags & MASK_TAIL_SWAP_DONE))
894  {
895  // Tail area restore is required and we have not yet written the relevant data page
896  // to swap area. Start the process of writing the data page to swap.
897  m_flags |= MASK_TAIL_SWAP_DONE;
898 
899  swap_sub_state_state_change(STATE_ERASE_SWAP);
900  }
901  else
902  {
903  // Tail area restore is required and we have already written the relevant data page
904  // to swap area. Proceed by restoring the tail area.
905  swap_sub_state_state_change(STATE_RESTORE_TAIL);
906  }
907  }
908  else
909  {
910  swap_sub_sm_exit_action_run();
911  }
912  }
913  else
914  {
915  // As operation request was rejected by the flash API reissue the request.
916  swap_sub_state_err_busy_process();
917  }
918 }

1149 static void tail_word_size_calculate(pstorage_size_t cmd_end_of_storage_address,
1150  pstorage_size_t end_of_storage_address)
1151 {
1152  // Two different cases to resolve when calculating correct size for restore tail section:
1153  // 1) End of storage area and command end area are in the same page.
1154  // 2) End of storage area and command end area are not in the same page.
1155 
1156  const uint32_t end_of_storage_area_page = end_of_storage_address /
1157  PSTORAGE_FLASH_PAGE_SIZE;
1158  const uint32_t command_end_of_storage_area_page = cmd_end_of_storage_address /
1159  PSTORAGE_FLASH_PAGE_SIZE;
1160 
1161  if (end_of_storage_area_page == command_end_of_storage_area_page)
1162  {
1163  //lint -e{573} suppress "Signed-unsigned mix with divide".
1164  m_tail_word_size = (end_of_storage_address - cmd_end_of_storage_address) / sizeof(uint32_t);
1165  }
1166  else
1167  {
1168  //lint -e{573} suppress "Signed-unsigned mix with divide".
1169  m_tail_word_size = (PSTORAGE_FLASH_PAGE_SIZE -
1170  (cmd_end_of_storage_address % PSTORAGE_FLASH_PAGE_SIZE)) /
1171  sizeof(uint32_t);
1172  }
1173 }

1178 static void clear_operation_execute(void)
1179 {
1180  const cmd_queue_element_t * p_cmd = &m_cmd_queue.cmd[m_cmd_queue.rp];
1181  const pstorage_block_t cmd_block_id = p_cmd->storage_addr.block_id;
1182 
1183  const pstorage_size_t block_size = m_app_table[p_cmd->storage_addr.module_id].block_size;
1184  const pstorage_size_t block_count = m_app_table[p_cmd->storage_addr.module_id].block_count;
1185  const pstorage_block_t block_base_id = m_app_table[p_cmd->storage_addr.module_id].base_id;
1186 
1187  const bool is_start_address_page_aligned = (cmd_block_id % PSTORAGE_FLASH_PAGE_SIZE) == 0;
1188 
1189  // Calculate the end (1 beyond allocated area) for complete storage area and to the area only
1190  // within scope of this command.
1191  const pstorage_block_t end_of_storage_address = block_base_id + (block_size * block_count);
1192  const pstorage_block_t cmd_end_of_storage_address = cmd_block_id + p_cmd->size + p_cmd->offset;
1193 
1194  // Zero tail to make sure no extra erase is done erroneously.
1195  m_tail_word_size = 0;
1196 
1197  // If the following is true no swap access is needed:
1198  // - 1st logical test covers the case of: clear/update 1 complete single page.
1199  // - 2nd logical test covers the case of:
1200  // 1) Clear/update last allocated page and page is not full (page can't be shared between
1201  // multiple clients so the end of the page is unused area).
1202  // 2) Clear/update all allocated storage.
1203  if ((is_start_address_page_aligned && (p_cmd->size == PSTORAGE_FLASH_PAGE_SIZE)) ||
1204  (is_start_address_page_aligned && (cmd_end_of_storage_address == end_of_storage_address) &&
1205  (p_cmd->offset == 0)) || (p_cmd->storage_addr.module_id == RAW_MODE_APP_ID))
1206  {
1207  // Nothing to put to the swap and we can just erase the pages(s).
1208 
1209  m_current_page_id = cmd_block_id / PSTORAGE_FLASH_PAGE_SIZE;
1210 
1211  sm_state_change(STATE_DATA_ERASE);
1212  }
1213  else
1214  {
1215  // Not all the blocks for the module can be cleared, we need to use swap page for storing
1216  // data temporarily.
1217 
1218  m_head_word_size = ((cmd_block_id + p_cmd->offset) % PSTORAGE_FLASH_PAGE_SIZE) /
1219  sizeof(uint32_t);
1220 
1221  const bool is_cmd_end_address_page_aligned = ((cmd_end_of_storage_address %
1222  PSTORAGE_FLASH_PAGE_SIZE) == 0);
1223  if ((cmd_end_of_storage_address != end_of_storage_address) &&
1224  !is_cmd_end_address_page_aligned)
1225  {
1226  // When command area is not equal to end of the storage allocation area and not ending
1227  // to page boundary there is a need to restore the tail area.
1228  tail_word_size_calculate(cmd_end_of_storage_address, end_of_storage_address);
1229  }
1230 
1231  sm_state_change(STATE_DATA_ERASE_WITH_SWAP);
1232  }
1233 }

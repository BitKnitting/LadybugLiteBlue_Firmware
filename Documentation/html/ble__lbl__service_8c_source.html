160 static void on_write(ble_lbl_t * p_lbl, ble_evt_t * p_ble_evt)
161 {
162  SEGGER_RTT_WriteString(0,"***---> IN ON-WRITE, COMMAND EVENT \n");
163 
164  ble_gatts_evt_write_t * p_evt_write = &p_ble_evt->evt.gatts_evt.params.write;
165  SEGGER_RTT_printf(0,"...command integer value: %d\n",p_evt_write->data[0]);
166  int calValue ;
167  if (p_evt_write->handle == p_lbl->control_char_handles.value_handle) {
168  switch (p_evt_write->data[0]) {
169  case resetPHcalValues:
170  case resetECcalValues:
171  ladybug_reset_calibration_values(p_evt_write->data[0]);
172  update_calibration_characteristic(p_lbl);
173  break;
174  //ask the ladybug to read the probe's value. The value is then written to flash
175  case calibratePH4:
176  case calibratepH7:
177  ladybug_update_calibration_value(p_evt_write->data[0],0); //the ECvalue is not needed so sending in a 0
178  update_calibration_characteristic(p_lbl);
179  break;
180  case calibrateEC1:
181  SEGGER_RTT_WriteString(0,"...calibrate EC1\n");
182  calValue = p_evt_write->data[2] << 8 | p_evt_write->data[1];
183  ladybug_update_calibration_value(p_evt_write->data[0],calValue);
184  update_calibration_characteristic(p_lbl);
185  SEGGER_RTT_printf(0,"...EC1 calibration solution value: %d\n", calValue);
186  break;
187  case calibrateEC2:
188  SEGGER_RTT_WriteString(0,"...calibrate EC2\n");
189  //get the calibration solution value typed in by the user. The units are µS/cm. The data type is Int16
190  calValue = p_evt_write->data[2] << 8 | p_evt_write->data[1];
191  SEGGER_RTT_printf(0,"...EC2 calibration solution value: %d\n",calValue);
192  ladybug_update_calibration_value(p_evt_write->data[0],calValue);
193  update_calibration_characteristic(p_lbl);
194  break;
195  case undoPH4:
196  case undoPH7:
197  SEGGER_RTT_WriteString(0,"...undo either pH4, pH7\n");
198  int16_t calValue = p_evt_write->data[1] | p_evt_write->data[2] << 8;
199  SEGGER_RTT_printf(0,"pH calibration value: %d\n",calValue);
200  ladybug_undo_pH_calibration(p_evt_write->data[0],calValue);
201  update_calibration_characteristic(p_lbl);
202  break;
203  case undoEC1:
204  case undoEC2:
205  SEGGER_RTT_WriteString(0,"...undo either EC1 (Vin and Vout), or EC2 (Vin and Vout)\n");
206  int16_t EC_Vin = p_evt_write->data[1] | p_evt_write->data[2] << 8;
207  int16_t EC_Vout = p_evt_write->data[3] | p_evt_write->data[4] << 8;
208  ladybug_undo_EC_calibration(p_evt_write->data[0], EC_Vin, EC_Vout);
209 
210  update_calibration_characteristic(p_lbl);
211  break;
212  case updatePHandEC:
213  SEGGER_RTT_WriteString(0,"update pH and EC\n");
214  update_measurement_characteristic(p_lbl);
215  break;
216  case updateBatteryLevel:
217  SEGGER_RTT_WriteString(0,"update battery level\n");
218  update_battery_level_characteristic(p_lbl);
219  break;
220  case updateDeviceName:
221  SEGGER_RTT_WriteString(0,"update device name\n");
222  display_bytes(&p_evt_write->data[1],p_evt_write->len);
223  update_device_name(&p_evt_write->data[1],p_evt_write->len);
224  break;
225  default:
226  SEGGER_RTT_WriteString(0,"Unknown control\n");
227  break;
228  }
229  }
